def somma(float a,b | out float result) : void {
	result << a + b;
}


def sottrai(float a,b | out float result) : void {

	result << a - b;

}


def molitplica(float a,b | out float result) : void {
	result << a * b;
}

def dividi(float a,b | out float result) : boolean{
